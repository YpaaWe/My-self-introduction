ABC083B - Some Sums 
                            # coding: utf-8
# Your code here!
N,a,b=map(int,input().split())
count=0
def digitSum(n):
    # 数値を文字列に変換
    s = str(n)
    # １文字ずつ数値化し配列にする。
    array = list(map(int, s))
    # 合計値を返す
    return sum(array)

for i in range(N):
    j=i+1
    J=digitSum(j)
    if J>=a and J<=b:
        count=count+j
print(count)
